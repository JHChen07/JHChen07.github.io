查找数组元素位置

题目描述
找出元素 item 在给定数组 arr 中的位置
输出描述
如果数组中存在 item，则返回元素在数组中的位置，否则返回 -1
示例1
输入 

[ 1, 2, 3, 4 ], 3

输出 

2

代码 

/* 需要考虑代码的鲁棒性，
首先需要判断支不支持 indexOf 函数，
如若不支持则采用顺序查找的方法，
特别注意的是如果查找不到需要返回 -1*/
function indexOf(arr, item) {
    if(arr.length==0) return;
    if(Array.prototype.indexOf) {
        return arr.indexOf(item);
    } else {
        for(var i=0; i<arr.length; i++) {
            if(arr[i]===item) {
                return i;
            }
        }
    }
    return -1;
}

// 常规解法
function sum(arr) {
    if(arr.length==0) return;
    var temp = 0;
    for(var i = 0; i < arr.length; i++) {
        temp += arr[i];
    }
    return temp;
}
// forEach 方法
function sum(arr) {
    if(arr.length==0) return;
    var temp = 0;
    arr.forEach(function(item) {
        temp += item;
    })
    return temp;
}
// reduce 方法
function sum(arr) {
    if(arr.length==0) return;
    var result = arr.reduce(function(pre, cur) {
        return pre + cur;
    })
    return result;
}

查找重复元素

题目描述

示例1

输出

代码 

function duplicates(arr) {
    if(arr.length==0) return null;
    var result = [];
    arr = arr.sort(function(v1, v2) {
        return v1 - v2;
    });
    for(var i=0; i<arr.length; i++) {
        if(arr[i]===arr[i+1]) {
            result.push(arr[i]);
            while(arr[i]===arr[i+1]) {
                ++i;
            }
        }
    }
    return result;
}

计时器

题目描述
实现一个打点计时器，要求

  1. 从 start 到 end（包含 start 和 end），每隔 100 毫秒 console.log 一个数字，每次数字增幅为 1
  2. 返回的对象中需要包含一个 cancel 方法，用于停止定时操作
  3. 第一个数需要立即输出

代码 

/*
此题要注意数据的改变以及返回的是对象
*/
function count(start, end) {
    console.log(start);
    var counting = setInterval(function() {
        if(start < end) {
            // 若此处没有进行自增操作，则打印的是同一个数
        console.log(start += 1);
        } else {
            clearInterval(counting);
        }
    }, 100);
    // 此处返回一个对象，如果直接返回cancel=function，则返回的是一个函数
    return {
        cancel: function() {
            clearInterval(counting);
        }
    }
}

流程控制

题目描述
实现 fizzBuzz 函数，参数 num 与返回值的关系如下：

  1. 如果 num 能同时被 3 和 5 整除，返回字符串 fizzbuzz
  2. 如果 num 能被 3 整除，返回字符串 fizz
  3. 如果 num 能被 5 整除，返回字符串 buzz
  4. 如果参数为空或者不是 Number 类型，返回 false
  5. 其余情况，返回参数 num

示例1
输入 

15

输出 

15

代码 

function fizzBuzz(num) {
    if (num === null || num === "") {
        return false;
    } else if (!isNaN(num)) {
        if (num % 3 === 0 && num % 5 === 0) {
            return "fizzbuzz";
        } else if (num % 3 === 0) {
            return "fizz";
        } else if (num % 5 === 0) {
            return "buzz";
        } else return num;
    } else {
        return false;
    }
}

属性遍历

题目描述
找出对象 obj 不在原型链上的属性(注意这题测试例子的冒号后面也有一个空格~)

  1. 返回数组，格式为 key: value
  2. 结果数组不要求顺序

示例1
输入 

var C = function() {this.foo = 'bar'; this.baz = 'bim';}; 
C.prototype.bop = 'bip'; 
iterate(new C());

输出 

["foo: bar", "baz: bim"]

代码 

function iterate(obj) {
    var result = [];
    // 可以直接对obj进行遍历属性
    for(var key in obj){
        // 判断属性是在实例中还是原型对象中
        if(obj.hasOwnProperty(key)) {
            // 用obj[key]访问属性值，还可以运用数组的迭代算法
            result.push(key + ': ' + obj[key]);
        }
    }
    return result;
}

  4. 正确的格式如：$1,023,032.03 或者 $2.03，错误的格式如：$3,432,12.12 或者 $34,344.3

示例1
输入 

'$20,933,209.93'

输出 

true

代码 

function isUSD(str) {
    return /^\$(0|[1-9]\d{0,2})(,\d{3})*(\.\d{2})?$/.test(str);
}

格式化数字

题目描述
给定数字或者字符串，将其格式化输出，要求如下：

  1. 若为字符串，则返回 -1；不为字符串则每个 3 位用 ’ (半角全角均可)隔开；
  2. 数字格式化为包含小数点后两位，如 23333 -> 23,333.00；
  3. 特殊情况：2e4 -> 20,000.00 -2333 -> -2,333.00；
  4. 结果以字符串格式输出。

代码 

function format(str) {
    if(isNaN(str)) return -1;
    str = Number(str).toFixed(2).toString();
    if(str.length<=6) {
        return str;
    }
    var temp = str.split('');
    var result = temp.slice(temp.length-3);
    var count = 0;
    for(var i=temp.length-4; i>=0; i--) {
        if(count<3) {
            result.unshift(temp[i]);
            count++;
        } else {
            result.unshift(',');
            result.unshift(temp[i]);
            count = 1;
        }
    }
    return result.join('');
}

数组去重

题目描述
数组中可能存在重复元素，将数组去重，返回一个元素不重复的数组。
代码 

function remove(arr) {
    if(arr && arr.length<=1) return arr;
    var result  = [];
    arr.forEach(function(item, index) {
        if(arr.indexOf(item)===index) {
            result.push(item);
        }
    })
    return result;
}
