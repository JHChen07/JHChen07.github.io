函数绑定

函数绑定作用：将某个函数指针以值的形式进行传递，同时指定该函数的执行环境。
函数绑定要创建一个函数，可以在特定的 this 环境中以指定参数调用另一个函数。函数绑定常和回调函数和事件处理程序一起使用，以便在把函数当成变量传递时保留执行环境。但是绑定函数需要更多内存和开销，谨慎使用。 

var handler = {
    message: 'Event handled',

    handleClick: function(event) {
        alert(this.message);
    }
};

var btn = document.getElementById('my-btn');
EventUtil.addHandler(btn, 'click', handler.handleClick); //undefined

上述代码没有保存 handler.handleClick 的执行环境，this 对象指向 DOM 按钮并非 handler（在 IE8 中，this 指向 window），闭包修正： 

var handler = {
    message: 'Event handled',

    handleClick: function(event) {
        alert(this.message);
    }
};

var btn = document.getElementById('my-btn');
EventUtil.addHandler(btn, 'click', function(event) {
    handler.handleClick(event);
});

JavaScript提供了将函数绑定到指定执行环境的函数 bind()，bind() 接收一个函数和一个环境作为参数，并返回一个在给定环境中调用给定函数的函数。并且将所有参数原封不动地传递。 

function bind(fn, context) {
    return function() {
        return fn.apply(context, arguments);
    }
}

上述代码在 bind() 中创建了一个闭包， 这个闭包使用 apply() 调用传入的函数，并给 apply() 传递 context 对象和参数。这里的 arguments 对象是内部函数的。当调用返回的函数时，它会在给定环境中调用被传入的函数并给出所有参数。 

var handler = {
    message: 'Event handled',

    handleClick: function(event) {
        alert(this.message + ':' + event.type);
    }
};

var btn = document.getElementById('my-btn');
EventUtil.addHandler(btn, 'click', bind(handler.handleClick,  handler));
// 直接调用 bind() 函数
// EventUtil.addHandler(btn, 'click', handler.handleClick.bind(handler));

函数柯里化

函数柯里化用于创建具有默认参数值的函数。它与函数绑定的区别是当函数被调用时，返回的函数可能还需要传入一些参数。相同之处是使用一个闭包返回一个函数。
动态创建柯里化函数的方式：调用另一个函数并为它传入要柯里化的函数和必要参数。 

function curry(fn) {
    var args = Array.prototype.slice.call(arguments, 1);
    return function() {
        var innerArgs = Array.prototype.slice.call(arguments);
        var finalArgs = args.concat(innerArgs);
        return fn.apply(null, finalArgs);
    }
}
function add(num1, num2) {
    return num1 + num2;
}
var curriedAdd = curry(add, 5);
alert(curriedAdd(3)); //8
/*
var curriedAdd = curry(add, 5, 12);
alert(curriedAdd()); //17
*/

函数柯里化还常常作为函数绑定的一部分包含在其中，构造出更复杂的 bind() 函数。 

function bind(fn, context) {
    // 此时 bind() 函数接受一个包裹函数和一个 object 对象，所以转化要从第 3 个开始
    var args = Array.prototype.slice.call(arguments, 2);
    return function() {
        var innerArgs = Array.prototype.slice.call(arguments);
        var finalArgs = args.concat(innerArgs);
        return fn.apply(context, finalArgs);
    }
}
var handler = {
    message: 'Event handled',

    handleClick: function(name, event) {
        alert(this.message + ':' + name + ':' + event.type);
    }
};

var btn = document.getElementById('my-btn');
EventUtil.addHandler(btn, 'click', bind(handler.handleClick,  handler, 'my-btn'));
// 直接调用 bind() 函数
// EventUtil.addHandler(btn, 'click', handler.handleClick.bind(handler, 'my-btn'));

上述例子实现了同时给 event 对象再额外给事件处理程序传递参数。
